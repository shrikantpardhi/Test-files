**JPA Specifications for Complex Filtering:**
```java
public class [ENTITY_NAME]Specifications {
    
    public static Specification<[ENTITY_NAME]> hasStatus(Status status) {
        return (root, query, criteriaBuilder) -> 
            status == null ? null : criteriaBuilder.equal(root.get("status"), status);
    }
    
    public static Specification<[ENTITY_NAME]> createdBetween(LocalDateTime start, LocalDateTime end) {
        return (root, query, criteriaBuilder) -> {
            if (start == null && end == null) return null;
            if (start == null) return criteriaBuilder.lessThanOrEqualTo(root.get("createdDate"), end);
            if (end == null) return criteriaBuilder.greaterThanOrEqualTo(root.get("createdDate"), start);
            return criteriaBuilder.between(root.get("createdDate"), start, end);
        };
    }
    
    public static Specification<[ENTITY_NAME]> hasRelatedEntityWithType(Type type) {
        return (root, query, criteriaBuilder) -> {
            if (type == null) return null;
            Join<[ENTITY_NAME], RelatedEntity> relatedJoin = root.join("relatedEntities");
            return criteriaBuilder.equal(relatedJoin.get("type"), type);
        };
    }
    
    public static Specification<[ENTITY_NAME]> hasMinimumRelatedCount(Integer minCount) {
        return (root, query, criteriaBuilder) -> {
            if (minCount == null) return null;
            
            Subquery<Long> subquery = query.subquery(Long.class);
            Root<[ENTITY_NAME]> subRoot = subquery.from([ENTITY_NAME].class);
            Join<[ENTITY_NAME], RelatedEntity> subJoin = subRoot.join("relatedEntities");
            
            subquery.select(criteriaBuilder.count(subJoin))
                    .where(criteriaBuilder.equal(subRoot.get("id"), root.get("id")));
            
            return criteriaBuilder.greaterThanOrEqualTo(subquery, minCount.longValue());
        };
    }
    
    public static Specification<[ENTITY_NAME]> searchByKeyword(String keyword) {
        return (root, query, criteriaBuilder) -> {
            if (keyword == null || keyword.trim().isEmpty()) return null;
            
            String pattern = "%" + keyword.toLowerCase() + "%";
            return criteriaBuilder.or(
                criteriaBuilder.like(criteriaBuilder.lower(root.get("name")), pattern),
                criteriaBuilder.like(criteriaBuilder.lower(root.get("description")), pattern)
            );
        };
    }
}
```

```java
                .timestamp(LocalDateTime.now())
                .status(HttpStatus.CONFLICT.value())
                .error("Data Conflict")
                .message("The operation conflicts with existing data")
                .build();
        return ResponseEntity.status(HttpStatus.CONFLICT).body(error);
    }
```
